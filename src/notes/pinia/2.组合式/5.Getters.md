```ts title="store.ts"
export const useMainStore = defineStore('main', () => {
  const count = ref(0)

  // 这里直接不演了, 直接用计算属性就可以!
  const moreCount = computed(() => {
    return count.value + 10
  })

  return { count, moreCount }
})
```
