```kotlin
fun main(){
  var a = "aaa"
  when (a) {
      "aaa" -> println("a is aaa")
      "bbb" -> println("a is bbb")
      else -> println("a is something else")
  }
  val status = when (a) {
      "aaa" -> 1
      else -> 2
  }
}
```
